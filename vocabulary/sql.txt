CREATE table dictionary (id  NUMBER(10) GENERATED by default on null as IDENTITY,
word VARCHAR2(70) NOT NULL,
canon_word VARCHAR2(70),
code VARCHAR2(200),
amount NUMBER(10));

CREATE table words (id  NUMBER(10) GENERATED by default on null as IDENTITY,
word VARCHAR2(70) NOT NULL,
id_word NUMBER(10),
corpus_id NUMBER(10));
